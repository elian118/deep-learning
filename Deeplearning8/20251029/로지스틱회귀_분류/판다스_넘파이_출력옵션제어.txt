# Dataframe 객체 예쁘게 출력하는 방법
from prettytable import PrettyTable  # 3.3.0 version 설치

def prinf_df(df):
    table = PrettyTable(['']+ list(df.columns))
    for row in df.itertuples():
        table.add_row(row)
    print(str(table))
    print()

prinf_df(df)  # df(데이터프레임) 객체를 테이블 형태로 알아서 예쁘게 출력
